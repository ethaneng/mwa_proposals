import json
import boto3

def lambda_handler(event, context):
    
    params = event["queryStringParameters"]
    bucketName = "mwa-proposals" ## CHANGE TO YOUR BUCKET NAME
    key = params["key"]

    try:
        s3 = boto3.resource('s3')
        s3.Object(bucketName, key).delete()
    except:
        return {
        'statusCode': 505,
        'body': json.dumps('Error deleting file from s3')
    }
    
    
    return {
        'statusCode': 200,
        'body': json.dumps('File deleted')
    }
